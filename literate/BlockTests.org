#+begin_src js
it("Basic block", () => {
    const token = Token.Blank.factory();
    const block = Block(token);
    expect(block.token).toEqual(token);
})
#+end_src

#+begin_src js
it("Basic tape", () => {
    const token = Token.Blank.factory();
    const block = Block(token);
    const tape = Tape();
    
    tape.append(block);
    
    expect(tape.cells.length).toBe(1);
    expect(tape.cells[0]).toEqual(block);
})
#+end_src
